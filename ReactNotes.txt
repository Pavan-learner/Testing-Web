  const [myStyle, setmyStyle] = useState({
    color: "black",
    backgroundColor: "white",
  })

  return (
    <div className="content" style={myStyle}>
      <h1>Hey i am new for react</h1>

      <div className="buttons">
        <button

        })}}

            backgroundColor:"white",
          })}}
        >Disable Dark Mode</button>
      </div>
